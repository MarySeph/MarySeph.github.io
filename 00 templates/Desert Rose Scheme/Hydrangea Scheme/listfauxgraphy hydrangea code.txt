function resizeImage(img) {
  if (img.style.height === "auto") {
    img.style.height = "140px";
    img.style.width = "140px";
    img.style.overflow = "hidden";
    img.style.borderRadius = "5px";   
    img.children[0].style.left = "50%";
    img.children[0].style.top = "50%";
    img.children[0].style.transform = "translateX(-50%) translateY(-50%)";
  } 
 else  {
    img.style.height = "auto";
    img.style.width = "auto";
    img.style.overflow = "all";
    img.style.borderRadius = "0px";
    img.children[0].style.left = "0px";
    img.children[0].style.top = "0px";
    img.children[0].style.transform = "none";
  }
}
